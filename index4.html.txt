<body>
  <h3>Multiplication Table Generator</h3>

  <!-- Labels and Inputs -->
  <label for="number">Enter a number:</label>
  <input type="number" id="number" placeholder="Enter a number">

  <label for="start">Start Value:</label>
  <input type="number" id="start" placeholder="Start of range">

  <label for="end">End Value:</label>
  <input type="number" id="end" placeholder="End of range">

  <!-- Button -->
  <button id="generateBtn" type="button">Generate Table</button>

  <!-- Error message -->
  <p id="errormessage"></p>

  <!-- Table output -->
  <table id="table"></table>

  <script>
    const numberInput = document.getElementById('number');
    const startInput = document.getElementById('start');
    const endInput = document.getElementById('end');
    const generateBtn = document.getElementById('generateBtn');
    const errorMessage = document.getElementById('errormessage');
    const table = document.getElementById('table');

    generateBtn.addEventListener('click', () => {
      const num = numberInput.value.trim();
      const start = startInput.value.trim();
      const end = endInput.value.trim();

      table.innerHTML = "";
      errorMessage.textContent = "";

      if (num === "" || start === "" || end === "") {
        errorMessage.textContent = "All fields are required!";
        return;
      }

      const n = parseInt(num);
      const s = parseInt(start);
      const e = parseInt(end);

      let content = "<tr><th>Expression</th><th>Result</th></tr>";
      for (let i = s; i <= e; i++) {
        content += `<tr><td>${n} x ${i}</td><td>${n * i}</td></tr>`;
      }
      table.innerHTML = content;
    });
  </script>
